
source global_vars.sh

options=' -i id_rsa -o StrictHostKeyChecking=no '

for p in $s1_port $s2_port $s3_port; do

  echo ''
  echo Doing port $p

  ssh $options root@127.0.0.1 -p $p 'mkdir -p install_scripts'
  rsync -av " ssh -p $p -i id_rsa "  remote_scripts/*  root@127.0.0.1:install_scripts/

done
