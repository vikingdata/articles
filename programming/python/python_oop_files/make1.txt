
cd /tmp
mkdir -p sample_package1

echo '
# load classed from files
from .mod1 import Print1
from .mod2 import Print2
from .mod3 import Print3a,Print3b
from .mod4 import Print4


  # Make Classes available when loading with *
__all__ = ["Print1", "Print2", "Print3a","Print3b", "Print4"]
' >  sample_package1/__init__.py

echo "

class Print1:
  def __init__(self):
    pass
    
  def print1(self):
    print ('executing print1 of object from Print1 class.')

" > sample_package1/mod1.py

echo "

class Print2:
  def __init__(self):
    pass

  def print2(self):
    print ('executing print2 of object from Print2 class.')

" > sample_package1/mod2.py

echo "

class Print3a:
  def __init__(self):
    pass

  def print3a(self):
    print ('executing print3a of object from Print3a class.')

class Print3a:
  def __init__(self):
    pass

  def print3b(self):
    print ('executing print3b of object from Print3b class.')

" > sample_package1/mod3.py

echo "
  # You can import Print3 from the package level because of __init__.py
from sample_package1 import Print3a
  # Or you can give the full path. 
from sample_package1.mod3 import Print3a

  # We are not going to import Print3b which will fail.
  # However, Print4 works at the package level. 

class Print4:
  def __init__(self):
    pass

  def print4(self):
    print ('executing print4 of object from Print4 class.')
    print ('Creating Print3 object in print4 method from class Print4')
    temp = Print3()
    print ('Executing print3 method in print4 method4')
    temp.print3()

    try:
        temp = Print3()
        temp.print3()
    except:
        print ("Failed creating print3 object in print4")

" > sample_package1/mod4.py


echo "

  # import objects from __all__ in __init_-.py
from sample_package1 import *

  # create objects
a = Print1()
b = Print2()
c1 = Print3a()
c2 = Print3b()
d = Print4()
  # Execute methods of objects
a.print1()
b.print2()
c1.print3a()
c2.print3b()
d.print4()
" > sample1.py

python3 sample1.py
